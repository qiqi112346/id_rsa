# static toPrimitive(subLayer, innerView)

将模型和注记图层对象转换为Primitive

参数

名称 类型 描述
subLayer Object

地形图层对象
innerView View

cesium的view对象
Primitive
示例
const url = 'http://webclient.smaryun.com:8089/igs/rest/services/Scene/DaYanTa-M3D/SceneServer'
const addSceneLayer = async (url) => {
  const layer = new IGSSceneLayer({
    url,
  })
  const loadedLayer = await layer.load()
  const subLayers = loadedLayer.activeScene.sublayers
  for (let i = 0; i < subLayers.length; i++) {
    switch (_subLayer.type) {
      case SceneSubLayerType.g3d:
      case SceneSubLayerType.label:
      case SceneSubLayerType.sceneServer:
        // 若是注记类型则要传入viewer对象实现注记避让效果
        const primitive = IGSSceneLayerUtil.toPrimitive(_subLayer, viewer)
        viewer.scene.primitives.add(primitive)
        viewer.zoomTo(primitive)
        break
    }
  }
}
addSceneLayer(url)

# static toTerrainProvider(subLayer)

将地形图层对象转换为MapGISTerrainProvider

参数

名称 类型 描述
subLayer Object

地形图层对象
MapGISTerrainProvider | undefined
示例
const url = 'http://webclient.smaryun.com:8089/igs/rest/services/Scene/TwTerrain/SceneServer'
const addSceneLayer = async (url) => {
  const layer = new IGSSceneLayer({
    url,
  })
  const loadedLayer = await layer.load()
  const subLayers = loadedLayer.activeScene.sublayers
  for (let i = 0; i < subLayers.length; i++) {
    switch (_subLayer.type) {
      case SceneSubLayerType.terrain:
        const provider = IGSSceneLayerUtil.toTerrainProvider(_subLayer)
        if (provider) {
          viewer.terrainProvider = provider
        }
        break
    }
  }
}
addSceneLayer(url)
